// Example of liquidifying a vNFT
async function
liquidifyVNFT(vNFT, account) {
  // Approve the vNFT for transfer
  await vNFT.approve(LIQUIDAO_ADDRESS, tokenId);

  // Calculate the premium based on lock time
  const lockTimeRemaining = await vNFT.getLockTimeRemaining(tokenId);
  const premium = calculatePremium(lockTimeRemaining);

  // Deposit the vNFT and receive liquid tokens
  const liquidTokens = await Liquidao.deposit(tokenId, premium);
  return liquidTokens;
}
